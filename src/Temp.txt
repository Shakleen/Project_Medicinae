CREATE TABLE BASIC_INFO(PATIENT_ID NUMBER,PATIENT_NAME VARCHAR2(100),AGE NUMBER,SEX VARCHAR2(25),ADDRESS VARCHAR2(200),ADMISSION_DATE DATE,CONSTRAINT PK_PATIENT_ID PRIMARY KEY (PATIENT_ID),CONSTRAINT CHK_SEX CHECK (SEX IN ('MALE', 'FEMALE')),CONSTRAINT CHK_AGE CHECK (AGE >= 0))
CREATE SEQUENCE SEQ_PATIENT_ID MINVALUE 1 START WITH 1 INCREMENT BY 1 CACHE 10
CREATE TABLE PATIENT_PHONE_NUMBERS(PATIENT_ID NUMBER,PHONE_NUMBER VARCHAR2(20),CONSTRAINT FK_PATIENT_ID FOREIGN KEY (PATIENT_ID) REFERENCES BASIC_INFO)
